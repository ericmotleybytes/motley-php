29  /// Call a protected or private function in a class under test.

118  /// Create the root temporary directory if it does not yet exist.
119  /// @return The absolute path to the root temporary directory.
120  public function createTmpRootDir() : string {
121  $chk = $this->chkE;
122  if(!file_exists($this->tmpRootDir)) {
123  $status = mkdir($this->tmpRootDir);

130  /// Create a temporary subdirectory under the root temporary directory.
131  /// @param $relSubDir - The optional relative name of the subdirectory.
132  /// If not specified, a randomly named subdirectory will be created.
133  /// @return The absolute path of the created subdirectory.
134  public function createTmpSubDir(string $relSubDir=null) {
135  $chk = $this->chkE;
136  if(is_null($relSubDir)) {

138  } else {

142  $status = mkdir($subDir);
143  if (!$chk->checkNotFalse($status,"mkdir $subDir failed.")) { return ""; }
144  $this->tmpDirsCreated[] = $subDir;

148  /// Create a temporary dummy file.
149  /// @param $filespec - absolute file path for temporary file.
150  /// However, it must be in a temporary directory we created.
151  /// @param $dummy - An optional string to be optionally repeated in the file.
152  /// @param $repeat - The number of times to repeat $dummy.
153  /// @return The absolute path of the file created.
154  public function createTmpDummyFile(
155  string $filespec, string $dummy="", int $repeat=0) {
156  // verify $dir is a dir we created, or the root tempdir.
157  $chk = $this->chkE;
158  $dir = dirname($filespec);
159  $tst = in_array($dir,$this->tmpDirsCreated);
160  if(! $chk->checkTrue($tst,"$dir not created during session.")) { return ""; }
161  $fh = fopen($filespec,"w");
162  if(! $chk->checkNotFalse($fh,"fopen $filespec failed.")) { return ""; }
163  $this->tmpFilesCreated[] = $filespec;
164  $repeatCnt=0;
165  while($repeatCnt<$repeat) {
166  $bytes = fwrite($fh,$dummy);
167  if(! $chk->checkNotFalse($bytes,"fwrite $filespec failed.")) { return ""; }
168  $repeatCnt++;
169  }
170  fclose($fh);
171  return $filespec;
172  }

186  /// Delete as much of tmp work as possible.
187  function cleanupTmp() {
188  // try to delete all the tmp files we know about.
189  foreach($this->tmpFilesCreated as $file) {
190  if(file_exists($file)) {
191  $status = unlink($file);
192  }
193  }
194  $this->tmpFilesCreated = array();
195  // try to completely delete each directory we created.
196  $dirs = array_reverse($this->tmpDirsCreated);
197  foreach($dirs as $dir) {

199  $this->recursiveRmdir($dir);

201  }
202  $this->tmpDirsCreated = array();
203  }

205  /// Recursively delete a directory and all contents.
206  /// @param $path - The top level directory to remove.
207  public function recursiveRmdir(string $path) {
208  if(!file_exists($path)) { return; }
209  $files = glob($path . '/*');
210  foreach ($files as $file) {
211  if (is_dir($file)) {
212  $this->recursiveRmdir($file);
213  } else {
214  unlink($file);
215  }
216  }
217  rmdir($path);
