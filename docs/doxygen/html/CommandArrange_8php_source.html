107  /// Try parsing command lines arguments.
108  /// @param $argv - The command line argument array.
109  /// @return TRUE if command line arguments match arrangement, else false.
110  public function parse(array $argv) : bool {
111  $this->argv = $argv;
112  $ddFound = false; // has "--" been found yet?
113  $this->rewindParsing();
114  $params = array_slice($argv,1);
115  $aComps = $this->getArranComps();
116  $param = array_shift($params);
117  $aComp = array_shift($aComps);
118  $match = null;
119  $reason = "";
120  while(true) { // note: break command used to stop looping
121  if($param===null) {
122  // No more command line args.
123  while($aComp!==null) {
124  // Check that remaining comps are fulfilled (optional or satisfied).
125  if($aComp->getIsFulfilled()===true) {
126  # fulfilled, we can skip this comp
127  $aComp = array_shift($aComps);
128  continue;
129  } else {
130  # mismatch!
131  $match = false;
132  $reason = "Not all required components fulfilled.";
133  break 2; // exit both while loops
134  }
135  }
136  // No more params or components.
137  // Apparently, we have a match!
138  $match = true;
139  $reason = "All requirements met.";
140  break;
141  } elseif($param=="--") {
142  // found a bare double dash
143  // skip fulfilled comps until validating double dash comp found.
144  $classNames = array('Motley\CommandDoubleDash');
145  $status = $this->skipFulfilledUntilClass(
146  $aComp,$aComps,$classNames,$param);
147  if($status===false) {
148  # could not find double dash component, mismatch!
149  $match = false;
150  $reason = "Could not find '--' component.";
151  break;
152  }
153  # $aComp is a validated double dash.
154  $ddFound = true;
155  $this->matchingComps[] = $aComp->getCompObj();
156  $param = array_shift($params); // next command line param
157  } elseif ($param!="-" and substr($param,0,1)=="-" and $ddFound==false) {
158  # long or short form switch
159  if(substr($param,0,2)=="--") {
160  $longForm = true; # --xxx long form
161  $param1 = $param;
162  $param2 = null;
163  $possible = null;
164  } else {
165  $longForm = false; # -x short form
166  // look ahead for possible switch arg
167  $possible = array_shift($params);
168  if($ddFound==false and !is_null($possible) and
169  CommandOpt::checkOptSwitch($possible)) {
170  # $possible is another switch, not a switch arg.
171  # undo look ahead.
172  array_unshift($params,$possible);
173  $possible = null;
174  $param1 = $param;
175  $param2 = null;
176  } else {
177  $param1 = $param;
178  $param2 = "$param $possible";
179  }
180  }
181  $paramsToTry = array();
182  if(!is_null($param2)) {
183  $paramsToTry[] = $param2;
184  }
185  $paramsToTry[] = $param1;
186  $classNames = array();
187  $classNames[] = 'Motley\CommandOpt';
188  $classNames[] = 'Motley\CommandOptGrp';
189  $aCompsSave = $aComps;
190  # skip fulfilled comps until validating opt or optgrp found
191  foreach($paramsToTry as $tryParam) {
192  $aComps = $aCompsSave;
193  $lastParam = $tryParam;
194  $status = $this->skipFulfilledUntilClass(
195  $aComp,$aComps,$classNames,$tryParam);
196  if($status) {
197  break;
198  }
199  }
200  if($status===false) {
201  # could not find validating opt or optgrp, mismatch!
202  $match = false;
203  $reason = "Option '$param' not expected here.";
204  break;
205  }
206  # $aComp is a validated option or option group.
207  if($lastParam===$param1) {
208  # we didn't need the option arg param, we should unshift it back.
209  if(!is_null($possible)) {
210  array_unshift($params,$possible);
211  }
212  }
213  $comp = $aComp->getCompObj();
214  if(is_a($comp,'Motley\CommandOptGrp')) {
215  # $comp is a CommandOptGrp
216  # save the embedded option that validated.
217  $this->matchingComps[] = $comp->getLastValidOption();
218  } else {
219  # $comp is a CommandOpt
220  $this->matchingComps[] = $comp;
221  }
222  $param = array_shift($params); // next command line param
223  } else {
224  # argument
225  $classNames = array('Motley\CommandArg');
226  $status = $this->skipFulfilledUntilClass(
227  $aComp,$aComps,$classNames,$param);
228  if($status===false) {
229  # could not find a validating arg, mismatch!
230  $match = false;
231  $reason = "Argument '$param' not expected here.";
232  break;
233  }
234  }
235  }
236  $this->matched = $match;
237  $this->matchingReason = $reason;
238  return $this->matched;
239  }

261  private function skipFulfilledUntilClass(&$aComp,&$aComps,$classNames,&$param) {
262  if (!is_array($classNames)) { $className = array($classNames); }
263  $result = false;
264  while($aComp!==null) {
265  $comp = $aComp->getCompObj();
266  foreach($classNames as $className) {
267  if(is_a($comp,$className)) {
268  # Found matching class name.
269  if($aComp->getIsRepeatable()) {
270  # Is a repeater, but it still must validate param.
271  $stat = $comp->validate($param);
272  if($stat===false) {
273  break; # out of foreach loop
274  }
275  } elseif($aComp->getIsFulfilled()===false) {
276  # is not yet fulfilled, but it still must validate.
277  $stat = $comp->validate($param);
278  if($stat===false) {
279  break; # out of foreach loop
280  }
281  } else {
282  # is a non-repeating and fulfilled component, we cannot use it.
283  break; # out of foreach loop.
284  }
285  if((!$aComp->getIsFulfilled()) or $aComp->getIsRepeatable()) {
286  return true;
287  }
288  # classname and validation are ok.
289  return true;
290  }
291  }
292  # Nothing found yet.
293  # If current comp is fulfilled, skip to the next one, else mismatch!
294  if($aComp->getIsFulfilled()) {
295  $aComp = array_shift($aComps);
296  } else {
297  return false;
298  }
299  }
300  # Went through all remainig components, but nothing matched param.
301  return false;
302  }
