15  /// Attempt to find source file from class name, and if found, include the file.
16  /// @param $class_name - The string class name of a class.
17  /// @return Returns false if no good file match found.
18  public static function autoload(string $class_name) {
19  // first replace back slashes with forward slashes.
20  $file_name = str_replace("\\","/",$class_name);
21  // add the expected file suffix.
22  $file_name = $file_name . ".php";
23  // look in parent directory of this source file, but using a sub-directory
24  // path driven my namespaces.
25  $candidate = __DIR__ . '/../' . $file_name;
26  $status = false;
27  if(file_exists($candidate)) {
28  $status = include_once($candidate);
29  } else {
30  // try to find something in the include_path.
31  $candidate = stream_resolve_include_path($file_name);
32  if($candidate!==false) {
33  // @codeCoverageIgnoreStart
34  $status = include_once($candidate);
35  // @codeCoverageIgnoreEnd
36  }
37  }
38  return $status;
39  }
