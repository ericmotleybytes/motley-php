33  /// Add an option to the option group.
34  /// If an option object with the same name already exists a warning
35  /// is issued and the option object if not added to the group.
36  /// @param $option - A CommandOpt object to add to the group.
37  /// @returns New number of options in group.
38  public function addOption(CommandOpt $option) : int {
39  $optionName = $option->getName();
40  foreach($this->options as $opt) {
41  if($optionName==$opt->getName()) {
42  $msg = "Option '$optionName' already in group.";
43  trigger_error($msg,E_USER_WARNING);
44  return count($this->options);
45  }
46  }
47  $switches = array();
48  foreach($option->getOptSwitches() as $switch) {
49  if(in_array($switch,$this->switches)) {
50  $msg = "Switch '$switch' already used in group.";
51  trigger_error($msg,E_USER_WARNING);
52  return count($this->options);
53  }
54  $switches[] = $switch;
55  }
56  $this->options[] = $option;
57  $this->switches = array_merge($this->switches,$switches);
58  return count($this->options);
59  }

107  /// Validate a switch parameter against a group of component options.
108  /// This overrides the parent validate function.
109  /// @param $param - A command line parameter(s).
110  /// @return TRUE is trhe param if valid for one of the options in the group.
111  /// Only one option should match because overlapping options (two options which
112  /// use the same switch) are prevented when adding options to the group.
113  public function validate(string $param) : bool {
114  $result = false;
115  $options = $this->getOptions();
116  foreach($options as $option) {
117  $status = $option->validate($param);
118  if($status===true) {
119  $result = true;

122  $this->saveLastParam($param,$result,$message);
123  break;
124  }
125  }
126  return $result;
127  }
