164  /// Test if an actual argument value meets the validation criteria
165  /// of this command line argument.
166  /// Also saves the value of the last argument validated and the
167  /// validation result.
168  /// @param $param - A string parameter, typically a specific token
169  /// from the command line.
170  /// @return TRUE if $param is valid, else FALSE.
171  public function validate(string $param) : bool {
172  if($param=="-") {
173  # replace "-" with default value
174  $param = $this->defaultValue;
175  }
176  $result = null;
177  // try to validate with literals
178  foreach($this->validLiterals as $literal) {
179  if ($param==$literal) {
180  $result = true;
181  $message = "'$param' matches valid literal.";
182  break;
183  }
184  }
185  if (is_null($result)) {
186  // try to validate with regular expressions
187  foreach($this->validRegExs as $regEx) {
188  $check = preg_match($regEx,$param);
189  if ($check===false) {
190  // @codeCoverageIgnoreStart
191  // Impossible path, but included for defensive programming.
192  continue;
193  // @codeCoverageIgnoreEnd
194  } elseif ($check==1) {
195  $result = true;
196  $message = "'$param' matches valid regex.";
197  break;
198  }
199  }
200  }
201  if (is_null($result)) {
202  // check if no literals or regexs required.
203  if(count($this->getValidLiterals())==0) {
204  if(count($this->getValidRegExs())==0) {
205  $result = true;
206  $message = "'$param' is ok (no special restrictions).";
207  }
208  }
209  }
210  if (is_null($result)) {
211  // there were literals or regexs, but none matched
212  $result = false;
213  $message = "'$param' did not match any literals or patterns.";
214  }
215  if($result===true) {
216  if($this->isFilename===true) {
217  $fileExists = file_exists($param);
218  if($this->fileMustExist==true) {
219  if ($fileExists!==true) {
220  $result = false;
221  $message = "'$param' does not exist.";
222  }
223  }
224  if($this->fileMustNotExist==true) {
225  if ($fileExists===true) {
226  $result = false;
227  $message = "'$param' already exists.";
228  }
229  }
230  }
231  }
232  $this->saveLastParam($param, $result, $message);
233  return $result;
234  }
