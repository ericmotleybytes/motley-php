10 /// Text formatter especially suited for usage explanations.
11 /// For example, man pages and --help screens.
12 /// This class implements a capability to wrap plain text not by words,
13 /// but rather by user-defined chunks of text. A "wrap" may occur between
14 /// chunks, but never within one. However a "cut" may happen if absolutely
15 /// needed, such as if the chunk length is longer than display width.
16 /// More traditional word wrap is still available.
17 /// It is also possible to place text at (or beyond) certain columns.
18 /// Additional features are also implemented, just as left and right
19 /// indenting, continuation indenting, multi-line column alignment,
20 /// maximum total length, automatic display column width detection,
21 /// and so forth.
22 class UsageFormatter {

61  /// Format some wrapable text.
62  /// @param $text - some wrapable text. Should not generally
63  /// have any embedded line-feeds, carriage returns, or tabs.
64  /// @param $col - column number (starting at 1) at which the
65  /// chunk is desired to be placed. If $col is too small or two
66  /// large for current indentations and display widths it will
67  /// be automatically adjusted. Zero means any column is ok.
68  /// @param $isMinCol - false (default) means chunk placed at exactly $col,
69  /// true means at of after.
70  public function formatText(
71  string $text, int $col=0, bool $isMinCol=false) {
72  $words = explode(" ",$text);
73  $wordCnt = 0;
74  foreach($words as $word) {
75  $wordCnt++;
76  if($wordCnt==1) {
77  $this->formatChunk($word,$col,$isMinCol);
78  } else {
79  $this->formatChunk($word,$col,true);
80  }
81  }
82  }

84  /// Format some non-wrapable text.
85  /// @param $chunk - a non-wrapable chunk of text. Should not generally
86  /// have any embedded line-feeds, carriage returns, or tabs.
87  /// @param $col - column number (starting at 1) at which the
88  /// chunk is desired to be placed. If $col is too small or two
89  /// large for current indentations and display widths it will
90  /// be automatically adjusted. Zero means any column is ok.
91  /// @param $isMinCol - false (default) means chunk placed at exactly $col,
92  /// true means at of after.
93  public function formatChunk(
94  string $chunk, int $col=0, bool $isMinCol=false) {
95  $lIndent = $this->getParam(self::LEFT_INDENT);
96  $rIndent = $this->getParam(self::RIGHT_INDENT);
97  $cIndent = $this->getParam(self::CONT_INDENT);
98  $maxLen = $this->getParam(self::MAX_LENGTH);
99  $width = $this->getColumnWidth();
100  # effective width is width less right indent, but it still
101  # must be at least left indent plus continue indent plus 1.
102  $effWidth = $width - $rIndent;
103  $effWidth = max($effWidth,$lIndent+$cIndent+1);
104  $eol = $this->getParam(self::EOL);
105  $space = " ";
106  # column position (if not 0) must be > total indent and
107  # also <= effective width
108  if ($col>0) {
109  $col = max($col,$lIndent+$cIndent+1);
110  $col = min($col,$effWidth);
111  }
112  $continued = false;
113  $effDelim = $space;
114  $lines = explode($eol,$this->textBuf);
115  $savedChunk = $chunk;
116  $sameCount = 0;
117  $chunk = str_replace("\n","",$chunk);
118  $chunk = str_replace("\r","",$chunk);
119  $chunk = str_replace("\t"," ",$chunk);
120  while(strlen($chunk)>0) {
121  // add defensive code to prevent accidental infinite looping
122  // @codeCoverageIgnoreStart
123  if($savedChunk==$chunk) {
124  $sameCount++;
125  if ($sameCount>3) {
126  $msg = "Infinite formatting loop detected.";
127  trigger_error($msg,E_USER_WARNING);
128  break;
129  }
130  } else {
131  $sameCount = 0;
132  }
133  // @codeCoverageIgnoreEnd
134  # begin processing
135  $last = count($lines) - 1; # last line index
136  // handle indenting
137  if($this->newBreak and strlen($lines[$last])==0) {
138  # very first chunk in continuation group
139  $lines[$last] .= str_repeat($space,$lIndent);
140  $effDelim = ""; # no need for delimiter
141  $this->newBreak = false;
142  } elseif(strlen($lines[$last])==0) {
143  # starting a continuation line
144  $lines[$last] .= str_repeat($space,$lIndent);
145  $lines[$last] .= str_repeat($space,$cIndent);
146  $effDelim = ""; # no need for delimiter
147  }
148  // handle delimiter
149  $charsRemaining = $effWidth - strlen($lines[$last]);
150  if($charsRemaining>=strlen($effDelim)) {
151  $lines[$last] .= $effDelim;
152  } else {
153  # add new line and loop again
154  $lines[] = "";
155  continue;
156  }
157  # if mincol is true, but not at col yet, turn off min flag
158  if($col>0 and $isMinCol===true) {
159  if(strlen($lines[$last])<=$col) {
160  $isMinCol = false;
161  }
162  }
163  if($col>0 and $isMinCol===false) {
164  # output at specific column requested
165  # if past column, add new line and loop.
166  if ( strlen($lines[$last]) >= $col) {
167  $lines[] = "";
168  continue;
169  }
170  # pad to column if needed
171  $padCharsNeeded = ($col-1) - strlen($lines[$last]);
172  if($padCharsNeeded>0) {
173  $lines[$last] .= str_repeat($space,$padCharsNeeded);
174  }
175  # add as much of $chunk as possible, cutting if needed.
176  $charsRemaining = $effWidth - strlen($lines[$last]);
177  if(strlen($chunk)<=$charsRemaining) {
178  # entire chunk fits
179  $lines[$last] .= $chunk;
180  $chunk = ""; # meets while loop exit condition
181  } else {
182  # force to do cut
183  $portion = substr($chunk,0,$charsRemaining);
184  $lines[$last] .= $portion;
185  $chunk = substr($chunk,$charsRemaining);
186  $lines[] = ""; # add another line and loop
187  continue;
188  }
189  } elseif($col>0 and $isMinCol===true) {
190  # output at or past specific column requested
191  $charsRemaining = $effWidth - strlen($lines[$last]);
192  if(strlen($chunk)<=$charsRemaining) {
193  $lines[$last] .= $chunk;
194  $chunk = ""; # meets while loop exit condition
195  } else {
196  $isMinCol = false; # now in at col mode
197  $lines[] = ""; # add another line and loop
198  continue;
199  }
200  } else {
201  # no specific output column requested
202  $charsRemaining = $effWidth - strlen($lines[$last]);
203  $contWidth = $effWidth - $this->getParam(self::CONT_INDENT);
204  $contWidth = $effWidth - $cIndent;
205  if(strlen($chunk)<=$charsRemaining) {
206  $lines[$last] .= $chunk;
207  $chunk = ""; # while exit condition
208  } elseif (strlen($chunk)<=$contWidth) {
209  # add new line and loop
210  $lines[] = "";
211  continue;
212  } else {
213  # forced to do cut
214  $portion = substr($chunk,0,$charsRemaining);
215  $lines[$last] .= $portion;
216  $chunk = substr($chunk,$charsRemaining);
217  $lines[] = ""; # add another line and loop
218  continue;
219  }
220  }
221  } # end of while loop
222  # trim possible trailing delim from lines
223  for($idx=0; $idx<count($lines); $idx++) {
224  $lines[$idx] = rtrim($lines[$idx]);
225  }
226  # convert line array into one big string and save it
227  $this->textBuf = implode($eol,$lines);
228  }

230  /// Get the chunk buffer as a (possibly multiline) string.
231  /// @return The resultant textual string.
232  public function getFormattedText() : string {
233  $result = $this->textBuf;
234  $maxlen = $this->params[self::MAX_LENGTH];
235  if($maxlen>0) {
236  if (strlen($result) > $maxlen) {
237  $result = substr($result,0,$maxlen);
238  }
239  }
240  return $result;
241  }
