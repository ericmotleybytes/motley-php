182  /// Test if a variable in an associative array.
183  /// @param $var - The variable to test.
184  /// @return TRUE if $var is an associative array, else false.
185  public static function isAssociativeArray($var) {
186  if (!is_array($var)) {
187  return false;
188  }
189  $compare = array_diff_key($var,array_keys(array_keys($var)));
190  if (count($compare)==0) {
191  return false;
192  } else {
193  return true;
194  }
195  }

197  /// Describe a variable type and value(s) as a length limited string.
198  /// @param $var - The variable to describe.
199  /// @param $maxLength - Optional, the maximum length of the result.
200  /// @return A string description of the variable, length limited.
201  public static function describeVar($var, int $maxLength=80) : string {
202  $result = "";
203  if(is_array($var)) {
204  if(self::isAssociativeArray($var)) {
205  $result .= '{';
206  $parts = array();
207  foreach($var as $k => $v) {
208  $parts[] = self::describeVar($k) . '=>' . self::describeVar($v);
209  }
210  $result .= implode(",",$parts);
211  $result .= '}';
212  } else {
213  $result .= '[';
214  $parts = array();
215  foreach($var as $v) {
216  $parts[] = self::describeVar($v,$maxLength);
217  }
218  $result .= implode(",",$parts);
219  $result .= ']';
220  }
221  } else {
222  $typ = gettype($var);
223  if($typ=="boolean") {
224  if($var===true) {
225  $typ .= "/true";
226  } else {
227  $typ .= "/false";
228  }
229  }
230  $str = strval($var);
231  $result .= "($typ)'$str'";
232  }
233  $result = trim($result);
234  if(strlen($result)>$maxLength) {
235  $result = substr($result,0,$maxLength-3) . "...";
236  }
237  return $result;
