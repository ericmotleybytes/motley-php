75  /// Generic user message display.
76  /// @param $msgType - The type of message. See class constants
77  /// VERBO_MSG, ERROR_MSG, DEBUG_MSG.
78  /// @param $prefix - String prepended to $msg.
79  /// @param $msg - The text message to display.
80  /// @param $lvl - Message level for possible filtering.
81  protected function genericMsg(
82  string $msgType, string $prefix, string $msg, int $lvl) : bool {
83  $chk = $this->chkE;
84  # validate message type.
85  $tst = in_array($msgType,$this->msgTypes);
86  if(! $chk->checkTrue($tst)) { return false; }
87  # check level
88  $maxLvl = $this->maxDispLvl[$msgType];
89  if($lvl > $maxLvl) {
90  return false;
91  }
92  # prepend prefix
93  $msg = "$prefix: $msg";
94  # make sure message ends with line terminator.
95  if(substr($msg,(0-strlen(PHP_EOL)))!=PHP_EOL) {
96  $msg .= PHP_EOL;
97  }
98  # make sure file handle is open
99  $dest = $this->msgDest[$msgType];
100  $fh = $this->msgFH[$msgType];
101  if(is_null($fh)) {
102  $fh = fopen($dest,'a+');
103  if(! $chk->checkNotFalse($fh)) { return false; }
104  $this->msgFH[$msgType] = $fh;
105  }
106  # output message
107  $byteCount = fwrite($fh,$msg);
108  # check fwrite status
109  if(! $chk->checkNotFalse($byteCount)) { return false; }
110  return true;
111  }
