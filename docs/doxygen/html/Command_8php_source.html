139  /// Get textual help information.
140  public function getHelp() : string {
141  $masterArgList = array();
142  $masterOptList = array();
143  $masterOptGrpList = array();
144  $fmt = $this->getUsageFormatter();
145  $fmt->clear();
146  $colWidth = $fmt->getColumnWidth();
147  $descCol = (int)($colWidth/3);
148  $descCol = min($descCol,40);
149  $descCol = max($descCol,4);
150  $fmt->setLeftIndent(0);
151  $fmt->setRightIndent(0);
152  $fmt->setContinueIndent(2);
153  $fmt->formatChunk("Name:");
154  $fmt->formatBreak();
155  $fmt->setLeftIndent(2);
156  $fmt->formatChunk($this->name . " -");
157  $fmt->formatText($this->description);
158  $fmt->formatBreak();
159  $fmt->setLeftIndent(0);
160  $fmt->formatChunk("Usage:");
161  $fmt->formatBreak();
162  $fmt->setLeftIndent(2);
163  foreach($this->arrangements as $arrangement) {
164  $fmt->formatChunk($this->getDisplayName());
165  $aComps = $arrangement->getArranComps();
166  foreach($aComps as $aComp) {
167  $comp = $aComp->getCompObj();
168  $isOpt = $aComp->getIsOptional();
169  $isRep = $aComp->getIsRepeatable();
170  if(is_a($comp,"\Motley\CommandArg")) {
171  $masterArgList[] = $comp;
172  $argString = $comp->getDisplayName();
173  if($isRep) {
174  $argString .= " ...";
175  }
176  if($isOpt) {
177  $argString = '[' . $argString . ']';
178  }
179  $fmt->formatChunk($argString);
180  } elseif(is_a($comp,"\Motley\CommandOpt")) {
181  $masterOptList[] = $comp;
182  $switchesString = $comp->getSwitchesString();
183  if($isOpt) {
184  $switchesString = '[' . $switchesString . ']';
185  }
186  $fmt->formatChunk($switchesString);
187  } elseif(is_a($comp,"\Motley\CommandOptGrp")) {
188  $masterOptGrpList[] = $comp;
189  $grpString = $comp->getDisplayName();
190  if($isOpt) {
191  $grpString = '[' . $grpString . ']';
192  }
193  $fmt->formatChunk($grpString);
194  } elseif(is_a($comp,"\Motley\CommandDoubleDash")) {
195  $ddString = $comp->getDisplayName();
196  if($isOpt) {
197  $ddString = '[' . $ddString . ']';
198  }
199  $fmt->formatChunk($ddString);
200  }
201  }
202  $fmt->formatBreak();
203  }
204  # output argument help (if any)
205  if(count($masterArgList)>0) {
206  $fmt->setLeftIndent(0);
207  $fmt->formatChunk("Arguments:");
208  $fmt->formatBreak();
209  $fmt->setLeftIndent(2);
210  foreach($masterArgList as $arg) {
211  $fmt->formatChunk($arg->getDisplayName());
212  $fmt->formatText($arg->getDescription(),$descCol);
213  $fmt->formatBreak();
214  }
215  }
216  # output option group help (if any)
217  if(count($masterOptGrpList)>0) {
218  foreach($masterOptGrpList as $optGrp) {
219  $fmt->setLeftIndent(0);
220  $fmt->formatChunk($optGrp->getDisplayName() . ":");
221  $fmt->formatBreak();
222  $fmt->setLeftIndent(2);
223  $opts = $optGrp->getOptions();
224  foreach($opts as $opt) {
225  $switchesString = $opt->getSwitchesString();
226  $fmt->formatChunk($switchesString);
227  $desc = $opt->getDescription();
228  $fmt->formatText($desc,$descCol);
229  $fmt->formatBreak();
230  }
231  }
232  }
233  # output options not in an option group
234  if(count($masterOptList)>0) {
235  $fmt->setLeftIndent(0);
236  $fmt->formatChunk("OtherOptions:");
237  $fmt->formatBreak();
238  $fmt->setLeftIndent(2);
239  foreach($masterOptList as $opt) {
240  $switchesString = $opt->getSwitchesString();
241  $fmt->formatChunk($switchesString);
242  $desc = $opt->getDescription();
243  $fmt->formatText($desc,$descCol);
244  $fmt->formatBreak();
245  }
246  }
247  return $fmt->getFormattedText();
248  }

256  /// Try to parse all defined arrangements against command line arguments.
257  /// @return An array of arrangements that match command line arguments.
258  public function parse() : array {
259  $argv = $this->argv;
260  $result = array();
261  foreach($this->arrangements as $arrangement) {
262  $matched = $arrangement->parse($argv);
263  if ($matched===true) {
264  $result[] = $arrangement;
265  }
266  }
