70  /// Generate a new, unique V4 GUID/UUID.
71  /// Modeled after example code
72  /// at <http://php.net/manual/en/function.com-create-guid.php>.
73  /// @returns A string with a hex representation of the 128 bit GUID.
74  public function generateGuid() : string {
75  $result = NULL;
76  // @codeCoverageIgnoreStart
77  // Impossible to force all fallback logic.
78  if (function_exists('openssl_random_pseudo_bytes') === true) {
79  // OSX/Linux generation method
80  $data = openssl_random_pseudo_bytes(16);
81  $data[6] = chr(ord($data[6]) & 0x0f | 0x40); // set version to 0100
82  $data[8] = chr(ord($data[8]) & 0x3f | 0x80); // set bits 6-7 to 10
83  if ($this->dashesFlag) {
84  $result = vsprintf('%s%s-%s-%s-%s-%s%s%s', str_split(bin2hex($data), 4));
85  } else {
86  $result = vsprintf('%s%s%s%s%s%s%s%s', str_split(bin2hex($data), 4));
87  }
88  } elseif (function_exists('com_create_guid') === true) {
89  // use Windows compatible generation method
90  $result = trim(com_create_guid(),'{}');
91  if ($this->dashesFlag===FALSE) {
92  $result = str_replace('-','',$result);
93  }
94  } else {
95  // generic fallback method
96  mt_srand((double)microtime() * 10000);
97  $charid = strtolower(md5(uniqid(rand(), true)));
98  if ($this->dashesFlag) {
99  $hyphen = '-';
100  } else {
101  $hyphen = '';
102  }
103  $result = substr($charid, 0, 8) . $hyphen
104  . substr($charid, 8, 4) . $hyphen
105  . substr($charid, 12, 4) . $hyphen
106  . substr($charid, 16, 4) . $hyphen
107  . substr($charid, 20, 12);
108  }
109  if ($this->upperFlag) {
110  $result = strtoupper($result);
111  } else {
112  $result = strtolower($result);
113  }
114  // @codeCoverageIgnoreEnd
115  return $result;
116  }
